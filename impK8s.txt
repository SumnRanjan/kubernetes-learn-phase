KUBERNETES IMPORTANT COMMANDS

1. Create Namespace
kubectl create namespace nginx

2. Get Namespaces
kubectl get ns

3. Create Resource from YAML
kubectl apply -f pod.yml
kubectl apply -f deployment.yml

4. Delete Resource from YAML
kubectl delete -f pod.yml
kubectl delete -f deployment.yml

5. Get Pods
kubectl get pods
kubectl get pods -n nginx
kubectl get pods -A

6. Get Deployments
kubectl get deployments -n nginx

7. Get ReplicaSets
kubectl get replicasets -n nginx

8. Describe Pod
kubectl describe pod/nginx-pod -n nginx

9. Describe Deployment
kubectl describe deployment/nginx-deployment -n nginx

10. Delete Pod
kubectl delete pod nginx -n nginx

11. Delete Deployment
kubectl delete deployment nginx-deployment -n nginx

12. Run Pod Directly
kubectl run nginx --image=nginx -n nginx

13. Exec Inside Pod
kubectl exec -it nginx -n nginx -- bash
kubectl exec -it nginx -n nginx -- sh

14. Check Logs
kubectl logs nginx -n nginx

15. Scale Deployment
kubectl scale deployment/nginx-deployment -n nginx --replicas=5

16. Update Image in Deployment
kubectl set image deployment/nginx-deployment nginx=nginx:1.25 -n nginx

17. Check All Resources
kubectl get all -n nginx

18. Check Nodes
kubectl get nodes

19. Check Cluster Info
kubectl cluster-info

20. Check Current Context
kubectl config current-context

21. Get Wide Output
kubectl get pods -o wide -n nginx

22. Watch Live Changes
kubectl get pods -n nginx -w

23. Port Forward Pod
kubectl port-forward pod/nginx 8080:80 -n nginx

24. Edit Deployment
kubectl edit deployment nginx-deployment -n nginx

25. Rollout Status
kubectl rollout status deployment/nginx-deployment -n nginx

26. Rollback Deployment
kubectl rollout undo deployment/nginx-deployment -n nginx
